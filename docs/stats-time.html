4.5.1 Unpaired t-test 

t_example <- tibble(
  x = rnorm(3, 10),
  y = rnorm(3, 14) 
) %>%
  pivot_longer(x:y)

t_example %>% 
  ggplot(aes(name, value)) +
  geom_jitter(width = 0.05) +
  expand_limits(y = 0)
t.test(value ~ name, data = t_example, var.equal = TRUE)
